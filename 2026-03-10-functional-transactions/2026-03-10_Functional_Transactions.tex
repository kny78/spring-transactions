    \begin{frame}

        \begin{columns}[t]
            \begin{column}{.55\textwidth}
                \begin{block}{@Transactional}
            @Transactional should begin a transaction before the method and commit after.

            Exceptions that leave the method will rollback the transaction.
                \end{block}
            \end{column}
            \begin{column}{.45\textwidth}
                \begin{block}{Problems}
                    \begin{enumerate}
                        \item Am I in a transaction?
            \item Need a \textbf{bean} to begin transaction
                        \item If my transaction fails, hard to handle
            \item Two Phase commit is over- and under-engineered
                    \end{enumerate}
                \end{block}
            \end{column}
        \end{columns}
    \end{frame}

    % TxManager
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
        \definecolor{bg}{rgb}{0.95,0.95,0.95}
    \defverbatim[colored]\txManager{
        \begin{kotlincode}
        class TxManager(val entityManagerFactory: EntityManagerFactory) {
            
            fun <R> autoCommitTx(block: (dbCtx: DbCtx) -> R): R {
                val em = entityManagerFactory.createEntityManager()

                // .use{} implies AutoCloseable
                return DbCtx(em).use { dbCtx ->
                    try {
                        block(dbCtx)
                    } catch (e: Exception) {
                        it.setRollbackOnly()
                        throw e
                    }
                }
            }
        }
    \end{kotlincode}
    }

    % DbCtx
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    \defverbatim[colored]\dbCtx{
        \begin{kotlincode}
        class DbCtx(val em: EntityManager) : AutoCloseable {
            val transaction: EntityTransaction = em.transaction
            init { transaction.begin() }

            val person: PersonDao by lazy { PersonDao(em) }
            val image: ImageDao by lazy { ImageDao(em) }
            
            override fun close() {
                when {
                    transaction == null -> println("init error")
                    transaction.rollbackOnly -> {
                        LOG.warn("Rolling back tx")
                        transaction.rollback()
                    }
                    else -> transaction.commit()
                }

                if (em.isOpen) em.close()
            }
        }
    \end{kotlincode}
    }

    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    % PersonDao
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    \defverbatim[colored]\personDao{
        \begin{kotlincode}
        class PersonDao(val em: EntityManager) {
            fun persist(person: Person): Person {
                em.persist(person)
                return person
            }

            fun list(): List<Person> {
                return em.createQuery("SELECT p from Person p", Person::class.java)
                    .resultList
            }
            
            fun byId(id: Long?): Person {
                return em
                    .createQuery("SELECT p from Person p where p.id=:id", Person::class.java)
                    .setParameter("id", id)
                    .singleResult
            }
            
            fun deleteAll() {
                em.createQuery("DELETE from Person")
                    .executeUpdate()
            }
        }
    \end{kotlincode}
    }
